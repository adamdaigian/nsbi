SELECT s.pricing_tier, DATE_TRUNC(m.churned_at, MONTH) AS churn_month,
  COUNT(*) AS accounts, SUM(m.mrr_lost) AS churn_mrr
FROM analytics.core.mrr_movements m
JOIN analytics.core.subscriptions s ON m.customer_id = s.customer_id
WHERE m.movement_type = 'churn' AND s.segment = 'Mid-Market'
AND m.churned_at >= '2024-04-01'
GROUP BY 1, 2 ORDER BY 2, 1
