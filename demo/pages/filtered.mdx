```sql
-- name: filtered_users
SELECT
  DATE_TRUNC('month', signup_date) as month,
  COUNT(*) as signups,
  SUM(CASE WHEN status = 'churned' THEN 1 ELSE 0 END) as churned,
  SUM(CASE WHEN status = 'active' THEN 1 ELSE 0 END) as active
FROM users
WHERE source = ${source}
GROUP BY month
ORDER BY month
```

```sql
-- name: filtered_kpis
SELECT
  COUNT(*) as total,
  SUM(CASE WHEN status = 'active' THEN 1 ELSE 0 END) as active,
  SUM(CASE WHEN status = 'churned' THEN 1 ELSE 0 END) as churned,
  ROUND(SUM(CASE WHEN status = 'churned' THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 1) as churn_rate
FROM users
WHERE source = ${source}
```

```sql
-- name: filtered_plans
SELECT
  plan,
  COUNT(*) as users
FROM users
WHERE source = ${source}
GROUP BY plan
ORDER BY users DESC
```

```sql
-- name: filtered_industries
SELECT
  industry,
  COUNT(*) as users
FROM users
WHERE source = ${source}
GROUP BY industry
ORDER BY users DESC
```

```sql
-- name: all_sources
SELECT
  DATE_TRUNC('month', signup_date) as month,
  source,
  COUNT(*) as signups
FROM users
GROUP BY month, source
ORDER BY month, source
```
