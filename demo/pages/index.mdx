```sql
-- name: kpis
SELECT
  (SELECT COUNT(*) FROM users) as total_users,
  (SELECT COUNT(*) FROM users WHERE status = 'active') as active_users,
  (SELECT ROUND(SUM(net_mrr), 0) FROM revenue) as total_net_mrr,
  (SELECT COUNT(*) FROM support_tickets WHERE status != 'resolved') as open_tickets
```

```sql
-- name: signups_by_month
SELECT
  DATE_TRUNC('month', signup_date) as month,
  COUNT(*) as signups
FROM users
GROUP BY month
ORDER BY month
```

```sql
-- name: mrr_trend
SELECT
  month,
  SUM(net_mrr) as net_mrr,
  SUM(new_mrr) as new_mrr,
  SUM(churn_mrr) as churn_mrr
FROM revenue
GROUP BY month
ORDER BY month
```

```sql
-- name: mrr_components
SELECT month, 'New MRR' as component, SUM(new_mrr) as amount FROM revenue GROUP BY month
UNION ALL
SELECT month, 'Expansion MRR', SUM(expansion_mrr) FROM revenue GROUP BY month
UNION ALL
SELECT month, 'Churn MRR', -SUM(churn_mrr) FROM revenue GROUP BY month
ORDER BY month, component
```

```sql
-- name: users_by_plan
SELECT
  plan,
  COUNT(*) as users,
  ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM users), 1) as pct
FROM users
GROUP BY plan
ORDER BY users DESC
```

```sql
-- name: users_by_source
SELECT
  source,
  COUNT(*) as users
FROM users
GROUP BY source
ORDER BY users DESC
```

```sql
-- name: recent_tickets
SELECT
  DATE_TRUNC('month', created_at) as month,
  category,
  COUNT(*) as tickets
FROM support_tickets
GROUP BY month, category
ORDER BY month
```
