```sql
-- name: rev_kpis
SELECT
  ROUND(SUM(net_mrr), 0) as total_net_mrr,
  ROUND(SUM(new_mrr), 0) as total_new_mrr,
  ROUND(SUM(expansion_mrr), 0) as total_expansion,
  ROUND(SUM(churn_mrr), 0) as total_churn
FROM revenue
```

```sql
-- name: mrr_by_plan
SELECT
  month,
  plan,
  net_mrr,
  active_subscriptions
FROM revenue
WHERE plan != 'free'
ORDER BY month, plan
```

```sql
-- name: mrr_net_trend
SELECT
  month,
  SUM(net_mrr) as net_mrr
FROM revenue
GROUP BY month
ORDER BY month
```

```sql
-- name: mrr_composition
SELECT month, 'New MRR' as component, SUM(new_mrr) as amount FROM revenue GROUP BY month
UNION ALL
SELECT month, 'Expansion MRR', SUM(expansion_mrr) FROM revenue GROUP BY month
UNION ALL
SELECT month, 'Churn MRR', -SUM(churn_mrr) FROM revenue GROUP BY month
ORDER BY month, component
```

```sql
-- name: plan_summary
SELECT
  plan,
  ROUND(SUM(net_mrr), 0) as total_mrr,
  SUM(active_subscriptions) as total_sub_months,
  ROUND(AVG(net_mrr), 0) as avg_monthly_mrr
FROM revenue
GROUP BY plan
ORDER BY total_mrr DESC
```

```sql
-- name: sub_status
SELECT
  status,
  COUNT(*) as count
FROM subscriptions
GROUP BY status
ORDER BY count DESC
```

```sql
-- name: churn_by_plan
SELECT
  s.plan,
  COUNT(*) as churned,
  ROUND(COUNT(*) * 100.0 / (
    SELECT COUNT(*) FROM subscriptions s2 WHERE s2.plan = s.plan
  ), 1) as churn_pct
FROM subscriptions s
WHERE s.status = 'cancelled'
GROUP BY s.plan
ORDER BY churn_pct DESC
```

```sql
-- name: ticket_stats
SELECT
  category,
  COUNT(*) as total,
  SUM(CASE WHEN status = 'resolved' THEN 1 ELSE 0 END) as resolved,
  ROUND(AVG(first_response_hours), 1) as avg_response_hrs,
  ROUND(SUM(CASE WHEN status = 'resolved' THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 1) as resolution_rate
FROM support_tickets
GROUP BY category
ORDER BY total DESC
```
