```sql
-- name: pipeline_kpis
SELECT
  COUNT(*) as total_deals,
  SUM(CASE WHEN stage = 'Closed Won' THEN amount ELSE 0 END) as revenue_won,
  ROUND(AVG(CASE WHEN stage NOT LIKE 'Closed%' THEN probability END), 1) as avg_probability,
  COUNT(CASE WHEN stage NOT LIKE 'Closed%' THEN 1 END) as open_deals
FROM deals
```

```sql
-- name: pipeline_by_stage
SELECT stage, COUNT(*) as deals, SUM(amount) as total_value
FROM deals
WHERE stage NOT LIKE 'Closed%'
GROUP BY stage
ORDER BY CASE stage
  WHEN 'Prospecting' THEN 1
  WHEN 'Qualification' THEN 2
  WHEN 'Proposal' THEN 3
  WHEN 'Negotiation' THEN 4
END
```

```sql
-- name: monthly_closed
SELECT
  DATE_TRUNC('month', close_date::DATE) as month,
  SUM(CASE WHEN stage = 'Closed Won' THEN amount ELSE 0 END) as won,
  SUM(CASE WHEN stage = 'Closed Lost' THEN amount ELSE 0 END) as lost
FROM deals
WHERE close_date IS NOT NULL
GROUP BY 1
ORDER BY 1
```

```sql
-- name: deals_by_source
SELECT source, COUNT(*) as deals, SUM(amount) as total_value
FROM deals
GROUP BY source
ORDER BY total_value DESC
```
