```sql
-- name: activity_summary
SELECT
  activity_type,
  COUNT(*) as count,
  ROUND(AVG(duration_minutes), 1) as avg_duration,
  SUM(duration_minutes) as total_minutes
FROM activities
GROUP BY activity_type
ORDER BY count DESC
```

```sql
-- name: daily_activities
SELECT
  activity_date::DATE as date,
  COUNT(*) as activities
FROM activities
GROUP BY 1
ORDER BY 1
```

```sql
-- name: activity_by_type_monthly
SELECT
  DATE_TRUNC('month', activity_date::DATE) as month,
  activity_type,
  COUNT(*) as count
FROM activities
GROUP BY 1, 2
ORDER BY 1
```

```sql
-- name: recent_activities
SELECT
  a.activity_date,
  r.rep_name,
  a.activity_type,
  d.deal_name,
  a.duration_minutes,
  a.notes
FROM activities a
JOIN reps r ON a.rep_id = r.rep_id
JOIN deals d ON a.deal_id = d.deal_id
ORDER BY a.activity_date DESC
LIMIT 100
```
