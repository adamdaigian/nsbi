```sql
-- name: event_kpis
SELECT
  COUNT(*) as total_events,
  COUNT(DISTINCT user_id) as unique_users,
  COUNT(DISTINCT session_id) as sessions,
  ROUND(COUNT(*) * 1.0 / COUNT(DISTINCT session_id), 1) as events_per_session
FROM events
```

```sql
-- name: events_by_type
SELECT
  event_type,
  COUNT(*) as count
FROM events
GROUP BY event_type
ORDER BY count DESC
```

```sql
-- name: events_by_feature
SELECT
  feature,
  COUNT(*) as total_events,
  COUNT(DISTINCT user_id) as unique_users
FROM events
GROUP BY feature
ORDER BY total_events DESC
```

```sql
-- name: events_over_time
SELECT
  DATE_TRUNC('month', ts) as month,
  event_type,
  COUNT(*) as count
FROM events
GROUP BY month, event_type
ORDER BY month, event_type
```

```sql
-- name: feature_over_time
SELECT
  DATE_TRUNC('month', ts) as month,
  feature,
  COUNT(*) as count
FROM events
GROUP BY month, feature
ORDER BY month, feature
```

```sql
-- name: top_users
SELECT
  e.user_id,
  u.plan,
  u.industry,
  COUNT(*) as events,
  COUNT(DISTINCT e.session_id) as sessions,
  COUNT(DISTINCT e.feature) as features_used
FROM events e
JOIN users u ON u.id = e.user_id
GROUP BY e.user_id, u.plan, u.industry
ORDER BY events DESC
LIMIT 15
```
