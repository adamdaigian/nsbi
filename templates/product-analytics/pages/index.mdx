```sql
-- name: engagement_kpis
SELECT
  COUNT(DISTINCT user_id) as total_users,
  COUNT(DISTINCT CASE WHEN session_date >= CURRENT_DATE - INTERVAL '1 day' THEN user_id END) as dau,
  COUNT(DISTINCT CASE WHEN session_date >= CURRENT_DATE - INTERVAL '30 days' THEN user_id END) as mau,
  ROUND(AVG(duration_seconds) / 60.0, 1) as avg_session_min
FROM sessions
```

```sql
-- name: dau_trend
SELECT
  session_date::DATE as date,
  COUNT(DISTINCT user_id) as dau
FROM sessions
WHERE session_date >= CURRENT_DATE - INTERVAL '90 days'
GROUP BY 1
ORDER BY 1
```

```sql
-- name: signups_by_month
SELECT
  DATE_TRUNC('month', signup_date::DATE) as month,
  COUNT(*) as signups
FROM users
GROUP BY 1
ORDER BY 1
```

```sql
-- name: users_by_plan
SELECT plan, COUNT(*) as users
FROM users
GROUP BY plan
ORDER BY users DESC
```

```sql
-- name: platform_breakdown
SELECT platform, COUNT(DISTINCT user_id) as users
FROM sessions
GROUP BY platform
ORDER BY users DESC
```
